Hangman Game Code

A classic Hangman game built with Python. This project demonstrates the use of loops, conditionals, and functions to create an interactive command-line game.


import random

def hangman():
    word_list = ["python", "java", "computer", "hacker", "developer"]
    word = random.choice(word_list)
    hanged_man = [
        "  +---+\n  |   |\n      |\n      |\n      |\n      |\n=========",
        "  +---+\n  |   |\n  O   |\n      |\n      |\n      |\n=========",
        "  +---+\n  |   |\n  O   |\n  |   |\n      |\n      |\n=========",
        "  +---+\n  |   |\n  O   |\n /|   |\n      |\n      |\n=========",
        "  +---+\n  |   |\n  O   |\n /|\\  |\n      |\n      |\n=========",
        "  +---+\n  |   |\n  O   |\n /|\\  |\n /    |\n      |\n=========",
        "  +---+\n  |   |\n  O   |\n /|\\  |\n / \\  |\n      |\n========="
    ]
    
    wrong_guesses = 0
    guessed_letters = []
    
    while wrong_guesses < len(hanged_man) - 1:
        display_word = ""
        for letter in word:
            if letter in guessed_letters:
                display_word += letter + " "
            else:
                display_word += "_ "
        
        print(hanged_man[wrong_guesses])
        print("Guessed letters:", " ".join(guessed_letters))
        print("Word:", display_word)
        
        if "_" not in display_word:
            print("Congratulations! You guessed the word!")
            break
            
        guess = input("Guess a letter: ").lower()
        
        if guess in guessed_letters:
            print("You already guessed that letter.")
        elif guess in word:
            guessed_letters.append(guess)
        else:
            guessed_letters.append(guess)
            wrong_guesses += 1
            
    if wrong_guesses == len(hanged_man) - 1:
        print(hanged_man[wrong_guesses])
        print("You lost! The word was:", word)

if __name__ == "__main__":
    hangman()
